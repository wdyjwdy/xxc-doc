- 含义：事件触发 n 秒后，执行回调，如果 n 秒内又触发，则重新计时
- 场景：连续按钮提交，实时搜索框
```js
function debounce(fn, wait) {
    let timeoutID = null
    
    return function() { // 不能使用 ()=>{}
	    if (timeoutID !== null) { clearTimeout(timeoutID) }
	    
        timeoutID = setTimeout(() => { // 不能使用 function(){}
	        fn.apply(this, arguments)
        }, wait)
    }
}
```
