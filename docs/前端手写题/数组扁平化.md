- 含义：将多维数组，变为一维数组
## flat()
```js
function flatten(arr) {
	return arr.flat(Infinity) // 数组深度未知
}
```
## 递归
```js
function flatten(arr) {
	output = []
	for (const val of arr) {
		if (Array.isArray(val)) {
			output = output.concat(flatten(val))
		} else {
			output.push(val)
		}
	}
	return output
}
```
